- ip/auth/login/ - method POST
	- fields:
		- username
		- password
	- return
		- access token
		- refresh token
- ip/auth/register/ - method POST
	- fields:
                - username
                - password
                - password2
                - email
                - first_name
                - last_name
        - return 
                - username
                - email
		- first_name
		- second_name
- ip/auth/login/refresh/ - method POST
	- fields:
		- refresh (token de refresh primit la login)
	- return:
		- access token
		- refresh token
- ip/auth/change_password/"id"/ - method PUT
	- fields:
		- password
		- password2
		- old_password
	- returnL
		- nothing if all ok
		- detail with error
- ip/auth/update_profile/"id"/ - method PUT
	- fields:
		- username
		- first_name
		- last_name
		- email
	- return:
		- username
		- first_name
		- last_name
		- email

	- needs authentification token
	- return
		- id
